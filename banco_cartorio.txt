
-- Tabela de Conjuge
CREATE TABLE Conjuge (
    ID_Conjuge INT AUTO_INCREMENT PRIMARY KEY,
    Nome VARCHAR(255),
    CPF CHAR(11),
    RG VARCHAR(20),
    Data_Nascimento DATE,
    Estado_Civil VARCHAR(50),
    Numero VARCHAR(10),
    Rua VARCHAR(255),
    Bairro VARCHAR(255),
    Cidade VARCHAR(255),
    UF CHAR(2),
    CEP CHAR(8),
    Telefone1 VARCHAR(15),
    Telefone2 VARCHAR(15)
);

-- Tabela de Casal
CREATE TABLE Casal (
    ID_Casal INT AUTO_INCREMENT PRIMARY KEY,
    ID_Conjuge1 INT,
    ID_Conjuge2 INT,
    Renda DECIMAL(10,2),
    FOREIGN KEY (ID_Conjuge1) REFERENCES Conjuge(ID_Conjuge),
    FOREIGN KEY (ID_Conjuge2) REFERENCES Conjuge(ID_Conjuge)
);

-- Tabela de Filho
CREATE TABLE Filho (
    ID_Filho INT AUTO_INCREMENT PRIMARY KEY,
    Nome VARCHAR(255),
    CPF CHAR(11),
    ID_Casal INT,
    Documento VARCHAR(255),
    FOREIGN KEY (ID_Casal) REFERENCES Casal(ID_Casal)
);

-- Tabela de Testemunhas
CREATE TABLE Testemunha (
    ID_Testemunha INT AUTO_INCREMENT PRIMARY KEY,
    Nome VARCHAR(255),
    CPF CHAR(11),
    ID_Casal INT,
    Copia_Identidade VARCHAR(255),
    Copia_Residencia VARCHAR(255),
    FOREIGN KEY (ID_Casal) REFERENCES Casal(ID_Casal)
);

-- Tabela de Documentos
CREATE TABLE Documentos (
    ID_Documentos INT AUTO_INCREMENT PRIMARY KEY,
    ID_Conjuge INT,
    Certidao_Nascimento VARCHAR(255),
    Copia_Identidade VARCHAR(255),
    Copia_Residencia VARCHAR(255),
    Certidao_Casamento VARCHAR(255),
    Certidao_Obito VARCHAR(255),
    Partilha_Bens VARCHAR(255),
    Comprovante_Renda VARCHAR(255),
    Data_Upload TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    Nome VARCHAR(255),
    FOREIGN KEY (ID_Conjuge) REFERENCES Conjuge(ID_Conjuge)
);
